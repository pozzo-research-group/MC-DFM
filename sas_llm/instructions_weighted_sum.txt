def objective_function(
    parameters,
    param_config,
    exp_data=None,
    mode="single",
    plot=False
):
    """
    parameters: normalized GA parameters
    param_config: dict defining parameter bounds
    exp_data: optional experimental data
    mode: 'single' or 'multiple'
    plot: optional plotting flag
    """

    # ---- Scale parameters ---- This changes depending on the user's requests#
    params = fitting.scale_parameters(parameters, param_config)

    d = params["d"]
    prop = params["prop"]
    background = params["background"]
    scale = params["scale"]

    # ---- Load structure once (could cache externally) ---- This is the coordinates of the building block #
    coordinates = Read_PDB.load_pdb_SANS('../Data/PDB/ferritin_pdb.pdb')
    lattice_coords = create_lattice_coords(d) #This function determines the arrangement of the building block. In this example they are dimers#
    n_pairwise = 10000000 #always set n_pairwise to 10000000#
    histogram_bins = 10000
    q = np.geomspace(0.012, 0.5, 3000)

    # ---- Monomer curve ---- Since our hypothesized sample is a mixture of protein monomers and dimers, we need to simulate the scattering curve of both#
    Iq_monomer = fitting.simulate_scattering(coordinates, q, histogram_bins, n_pairwise, mode='single')
    # ---- Simulate dimer ---- This simulates the dimer scattering curve#
    Iq_dimer = fitting.simulate_scattering_lattice(coordinates, lattice_coords, q, histogram_bins, n_pairwise, mode='single')
    # ---- Invariants ---- In SAXS we need to rescale the curves based on their invariants. This is done by dividing each curve by their invariant and the multiplying by the number of scattering objects.#
    Inv_monomer = fitting.invariant(np.column_stack((q, Iq_monomer)))
    Inv_dimer = fitting.invariant(np.column_stack((q, Iq_dimer)))
    #Number of scattering subunits#
    N_subunits_monomer = 1
    N_subunits_dimer = lattice_coords.shape[0]
    #Scale the intensities by the invariant and the number of scattering units or the volume# 
    Iq_monomer_scaled = fitting.scale_intensity(Iq_monomer, Inv_monomer, N_subunits_monomer, prop) #Do not add any keyword arguments to this function#
    Iq_dimer_scaled = fitting.scale_intensity(Iq_dimer, Inv_dimer, N_subunits_dimer, 1-prop) #Do not add any keyword arguments to this function#
    # ---- Mixture ---- This defines the total scatttering, which is composed of the weighted sum of the dimer intensity and monomer intensity.#
    I_q_all = (Iq_monomer_scaled + Iq_dimer_scaled)* scale + background
    model = np.column_stack((q, I_q_all))
    # ---- Match q-grid ----#
    model = fitting.convert_data(exp_data, model)
    # ---- Compute error ----#
    error = np.mean(np.abs(np.log10(np.abs(exp_data[:, 1])) - np.log10(np.abs(model[:, 1]))))

    return -error, model
